# Write your own code
datasets_filename = ['data_quiz/wdi_co2_2005.csv', 'data_quiz/wdi_co2_2010.csv', 'data_quiz/wdi_co2_2015.csv']

index = 0
for dataset_filename in datasets_filename:
    dataset = pd.read_csv(dataset_filename)
    co2_gasfuel_kt = dataset["co2_gasfuel_kt"]
    co2_liquidfuel_kt = dataset["co2_liquidfuel_kt"]

    co2_total_kt = co2_gasfuel_kt + co2_liquidfuel_kt
    #plot histogram of co2_total_kt
    plt.hist(co2_total_kt)
    plt.title("Total co2 emissions from " + dataset_filename)
    plt.xlabel("Total co2 emissions")
    plt.ylabel("Frequency")
    plt.show()
    index += 1
